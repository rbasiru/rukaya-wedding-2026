function enter(){
document.querySelector(".left").style.transform="translateX(-100%)";
document.querySelector(".right").style.transform="translateX(100%)";
setTimeout(()=>{document.getElementById("opening").style.display="none";},1500);
document.getElementById("music").play();
}

function toggleMusic(){
let m=document.getElementById("music");
let i=document.getElementById("musicIcon");
if(m.paused){m.play();i.className="fa-solid fa-music";}
else{m.pause();i.className="fa-solid fa-volume-xmark";}
}

const target=new Date("March 29, 2026 10:00:00").getTime();
setInterval(()=>{
let now=new Date().getTime();
let diff=target-now;
document.getElementById("d").innerHTML=Math.floor(diff/(1000*60*60*24));
document.getElementById("h").innerHTML=Math.floor((diff%(1000*60*60*24))/(1000*60*60));
document.getElementById("m").innerHTML=Math.floor((diff%(1000*60*60))/(1000*60));
document.getElementById("s").innerHTML=Math.floor((diff%(1000*60))/1000);
},1000);
